Infix to postfix
#include <stdio.h>
#include <ctype.h>

char stack[100];
int top = -1;

void push(char x) {
    stack[++top] = x;
}

char pop() {
    if (top == -1)
        return -1;
    else
        return stack[top--];
}

int priority(char x) {
    if (x == '(')
        return 0;
    if (x == '+' || x == '-')
        return 1;
    if (x == '*' || x == '/')
        return 2;
    if (x == '^')
        return 3;
    return 0;
}

int main() {
    char exp[100];
    char *e, x;

    printf("Enter the infix expression: ");
    scanf("%s", exp);

    e = exp;
    printf("Postfix expression: ");
    while (*e != '\0') {
        if (isalnum(*e)) {
            // Operand, print directly
            printf("%c", *e);
        } else if (*e == '(') {
            push(*e);
        } else if (*e == ')') {
            while ((x = pop()) != '(')
                printf("%c", x);
        } else {
            while (top != -1 && priority(stack[top]) >= priority(*e)) {
                printf("%c", pop());
            }
            push(*e);
        }
        e++;
    }

    // Pop remaining operators from stack
    while (top != -1) {
        printf("%c", pop());
    }

    printf("\n");
    return 0;
}
