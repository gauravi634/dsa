Circular queue
#include <stdio.h>
#include <stdlib.h>
#define max 5

int rear = -1, front = -1, ch, i, a[max];

void insert()
{
    int data;
    if ((rear == max - 1 && front == 0) || (rear + 1 == front))
    {
        printf("\nQueue is Full");
    }
    else
    {
        printf("\nEnter data: ");
        scanf("%d", &data);
        if (front == -1)
        {
            front = rear = 0;
        }
        else if (rear == max - 1)
        {
            rear = 0;
        }
        else
        {
            rear++;
        }
        a[rear] = data;
        printf("Data entered successfully.\n");
    }
}

void delete()
{
    if (front == -1)
    {
        printf("\nQueue is empty.\n");
    }
    else
    {
        printf("\nElement deleted: %d\n", a[front]);
        if (rear == front)
        {
            rear = front = -1;
        }
        else if (front == max - 1)
        {
            front = 0;
        }
        else
        {
            front++;
        }
    }
}

void display()
{
    if (rear == -1 && front == -1)
    {
        printf("\nQueue is empty\n");
    }
    else
    {
        printf("\nElements are:\n");
        if (rear < front)
        {
            for (i = front; i < max; i++)
            {
                printf("a[%d] = %d\n", i, a[i]);
            }
            for (i = 0; i <= rear; i++)
            {
                printf("a[%d] = %d\n", i, a[i]);
            }
        }
        else
        {
            for (i = front; i <= rear; i++)
            {
                printf("a[%d] = %d\n", i, a[i]);
            }
        }
    }
}

int main()
{
    int cont = 1;
    while (cont == 1)
    {
        printf("\n1. Insert \n2. Delete \n3. Display \n4. Exit");
        printf("\nEnter your choice: ");
        scanf("%d", &ch);

        switch (ch)
        {
        case 1:
            insert();
            break;
        case 2:
            delete();
            break;
        case 3:
            display();
            break;
        case 4:
            exit(0);
        default:
            printf("\nWrong Choice");
        }

        printf("\nDo you want to continue? (1 = Yes / 0 = No): ");
        scanf("%d", &cont);
    }

    return 0;
}
