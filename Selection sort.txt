Selection sort
#include<stdio.h> 
int main() 
{ 
    int a[5],i,j,n,temp; 
     
    printf("Enter the size of array"); 
    scanf("%d",&n); 
     
    printf("Enter the array element"); 
    for(i=0;i<n;i++) 
    { 
        scanf("%d",&a[i]); 
    } 
    for(i=0;i<n;i++) 
    { 
        for(j=i+1;j<n;j++) 
        { 
            if(a[i]>a[j]) 
            { 
                temp=a[i]; 
                a[i]=a[j]; 
                a[j]=temp; 
            } 
        } 
    } 
    printf("Sorted array=\n"); 
    for(i=0;i<n;i++) 
    { 
        printf("\n%d",a[i]); 
    } 
return 0; 
}
